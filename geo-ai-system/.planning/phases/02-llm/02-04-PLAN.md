```typescript
import { request } from '@/utils/request';
import type { LLMConfig, LLMTestResult } from '@/types/llm';

export const llmApi = {
  // Get current tenant's LLM config
  getLLMConfig: () => request.get<LLMConfig>('/api/v1/llm/config'),
  
  // Save/update LLM config
  saveLLMConfig: (config: LLMConfig) => request.post<void>('/api/v1/llm/config', config),
  
  // Test LLM connection
  testConnection: (config: LLMConfig) => request.post<LLMTestResult>('/api/v1/llm/config/test', config),
  
  // Delete LLM config
  deleteLLMConfig: () => request.delete<void>('/api/v1/llm/config'),
};
```

Create frontend/src/stores/llmStore.ts with Pinia store:
```typescript
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import { llmApi } from '@/api/llm';
import type { LLMConfig } from '@/types/llm';

export const useLLMStore = defineStore('llm', () => {
  const config = ref<LLMConfig | null>(null);
  const loading = ref(false);
  const error = ref<string | null>(null);

  const isConfigured = computed(() => config.value?.enabled === true);

  const fetchConfig = async () => {
    loading.value = true;
    error.value = null;
    try {
      const response = await llmApi.getLLMConfig();
      config.value = response.data;
    } catch (err: any) {
      error.value = err.message || 'Failed to fetch LLM config';
    } finally {
      loading.value = false;
    }
  };

  const saveConfig = async (newConfig: LLMConfig) => {
    loading.value = true;
    error.value = null;
    try {
      await llmApi.saveLLMConfig(newConfig);
      config.value = newConfig;
    } catch (err: any) {
      error.value = err.message || 'Failed to save LLM config';
      throw err;
    } finally {
      loading.value = false;
    }
  };

  const deleteConfig = async () => {
    loading.value = true;
    error.value = null;
    try {
      await llmApi.deleteLLMConfig();
      config.value = null;
    } catch (err: any) {
      error.value = err.message || 'Failed to delete LLM config';
      throw err;
    } finally {
      loading.value = false;
    }
  };

  return {
    config,
    loading,
    error,
    isConfigured,
    fetchConfig,
    saveConfig,
    deleteConfig,
  };
});
```

| Frontend → User | API key displayed in password field (masked) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-14 | Information Disclosure | API key visible in browser console | mitigate | Use password input type for apiKey field, never log full key in console |
| T-02-15 | Information Disclosure | API key in network request | mitigate | HTTPS enforced by Nginx (Phase 7), API key in request body (not URL) |
| T-02-16 | Tampering | User modifies API key in browser | accept | Frontend validation only; backend validates and encrypts key before storage |
| T-02-17 | Elevation of Privilege | User accesses other tenant's config | mitigate | Backend enforces tenant_id from JWT token, frontend only shows current tenant's config |
